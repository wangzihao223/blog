#include <algorithm>
#include <climits>
#include <cstdlib>
#include <cstring>
#include <limits.h>
#include <queue>
#include <stdio.h>
#include <utility>
#include <vector>
typedef long long ll;
#define INF (1LL << 60)

int N, M, Q;
ll dist[305][305];
int rm[200005];

int m_arr[200000][3];

int main() {
  scanf("%d %d %d", &N, &M, &Q);
  for (int i = 1; i <= N; i++) {
    for (int j = 1; j <= N; j++) {
      dist[i][j] = INF;
      if (i == j) {
        dist[i][j] = 0;
      }
    }
  }
  for (int i = 0; i < M; i++) {
    int a, b, w;
    scanf("%d %d %d", &a, &b, &w);
    dist[a][b] = w;
    dist[b][a] = w;
    m_arr[i][0] = a;
    m_arr[i][1] = b;
    m_arr[i][2] = w;
  }

  std::vector<std::vector<std::pair<int, int>>> query;
  query.push_back({});
  int index = 0;
  for (int i = 0; i < Q; i++) {
    int t;
    scanf("%d", &t);
    if (t == 2) {
      int a, b;
      scanf("%d %d", &a, &b);
      query[index].push_back(std::pair<int, int>(a, b));
    } else {
      int idx;
      scanf("%d", &idx);
      idx--;
      index++;
      query.push_back(std::vector<std::pair<int, int>>());
      rm[index] = idx;

      dist[m_arr[idx][0]][m_arr[idx][1]] = INF;
      dist[m_arr[idx][1]][m_arr[idx][0]] = INF;
    }
  }
  for (int k = 1; k <= N; k++) {
    for (int i = 1; i <= N; i++) {
      for (int j = 1; j <= N; j++) {
        if (dist[i][k] < INF && dist[k][j] < INF) {
          dist[i][j] = std::min(dist[i][j], dist[i][k] + dist[k][j]);
        }
      }
    }
  }
  ll tmp[200000];
  int n_size = 0;

  int start = query.size();
  for (int i = start - 1; i >= 0; i--) {
    int size = query[i].size();
    for (int j = size - 1; j >= 0; j--) {
      tmp[n_size] = dist[query[i][j].first][query[i][j].second];
      n_size++;
    }
    int u, v, w;
    u = m_arr[rm[i]][0];
    v = m_arr[rm[i]][1];
    w = m_arr[rm[i]][2];

    if (i == 0) {
      continue;
    }
    for (int j = 1; j <= N; j++) {
      for (int k = 1; k <= N; k++) {
        if (dist[j][u] < INF && dist[v][k] < INF) {
          dist[j][k] = std::min(dist[j][k], dist[j][u] + w + dist[v][k]);
        }
      }
    }
    for (int j = 1; j <= N; j++) {
      for (int k = 1; k <= N; k++) {
        if (dist[j][v] < INF && dist[u][k] < INF) {
          dist[j][k] = std::min(dist[j][k], dist[j][v] + w + dist[u][k]);
        }
      }
    }
  }
  for (int i = n_size - 1; i >= 0; i--) {
    if (tmp[i] == INF) {
      tmp[i] = -1;
    }
    printf("%lld\n", tmp[i]);
  }

  return 0;
}
