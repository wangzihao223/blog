
给定长度为N的数列 $A = (A_1, A_2, .. A_N)$ 和 $B=(B_1, B_2, ... B_N)$ . 现从$\{1, 2, ... N\}$中选出一个大小为K的子集S, 使得下式取值最小,求出该最小值:

$$(\max \limits_{i \in S}A_i) \times (\sum \limits_{i\in S}B_i)$$
```cpp
#include <stdio.h>
#include <cstring>
#include <math.h>
#include <queue>
#include <climits>
typedef long long ll;
typedef struct Pair{
	int a;
	int b;
}Pair;

int compare(const void* a, const void*b ){
	Pair* a1 = (Pair*)a;
	Pair* b1 = (Pair*)b;
	if(a1->a < b1->a) return -1;
	if(a1->a == b1->a) return 0;
	return 1;
}
typedef std::priority_queue<int> pq;


int main(){
	int T;
	scanf("%d", &T);
	for(int t=0; t<T; t++){
		int N, K;
		scanf("%d%d", &N, &K);
		Pair *p = (Pair*)malloc(sizeof(Pair)*N);
		for(int k=0; k<N; k++){
			scanf("%d", &p[k].a);
		}
		for(int k=0; k<N; k++){
			scanf("%d", &p[k].b);
		}
		qsort(p, N, sizeof(Pair), compare);
		pq heap;
		ll sum = 0;
		ll min_ans = LONG_LONG_MAX;
		for(int i=0; i<N; i++){
			Pair e =  p[i];
			if((int)heap.size() < K-1){
				heap.push(e.b);
				sum += e.b;
				continue;
			}
			if((int)heap.size() == K-1){
				min_ans =  std::min((sum + e.b)*e.a, min_ans);
				heap.push(e.b);
				sum += e.b;
			}

			if((int)heap.size() > K-1){
				int b1 = heap.top();
				heap.pop();
				sum -= b1;
			}
		}
		printf("%lld\n", min_ans);
	}
	return 0;
}
```
